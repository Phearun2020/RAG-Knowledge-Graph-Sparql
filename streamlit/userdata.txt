#!/bin/bash

# Legacy standalone user-data helper. The CloudFormation template in iac/main.yaml
# is the primary deployment path and fetches streamlit/main.py from S3.

sudo dnf install -y python3 python3-pip nodejs npm awscli

sudo -u ec2-user bash << 'EOF'
mkdir -p ~/streamlit

# Replace <YOUR_BUCKET> if you use this file directly outside CloudFormation.
aws s3 cp s3://<YOUR_BUCKET>/streamlit/main.py ~/streamlit/main.py

cat << 'REQ' > ~/streamlit/requirements.txt
boto3==1.43.18
streamlit==1.58.0
langchain==1.3.2
langchain-aws==1.5.0
langchain-ollama==1.1.0
langchain-openai==1.2.2
requests==2.34.2
REQ

python3 -m pip install --user -r ~/streamlit/requirements.txt
echo 'export PATH="$PATH:$HOME/.local/bin"' >> ~/.bashrc
source ~/.bashrc

sudo npm install pm2@latest -g
pm2 start "streamlit run ~/streamlit/main.py --server.port 8501"
pm2 startup systemd
pm2 save
pm2 install pm2-logrotate
EOF
